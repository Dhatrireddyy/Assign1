	\textbf{Problem 3.10.(c)} Without explicitly solving for X(z), find the region of convergence of the z-transform of each of the following sequences and determine whether the Fourier transform converges:
	\begin{align}
		x[\ n]\ = 2^{n}u[\ -n]\ 
	\end{align}
	
	\solution 
	\begin{align}
	    x[\ n]\ &= 2^{n}u[\ -n]\ \\
	    &= {\cbrak{1/2}}^{-n}u[\ -n]\ \\
	    x[\ -n]\ & \iff X\brak{\frac{1}{z}} 
	\end{align}
	
	${\cbrak{1/2}}^{n}u[\ n]\  \implies $ 
	ROC is $ \abs{z} > \frac{1}{2}$ \\
	
	${\cbrak{1/2}}^{-n}u[\ -n]\  \implies $ 
	ROC is $ \abs{z} < 2$
